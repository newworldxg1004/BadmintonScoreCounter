<button onclick="reset()">Reset</button>


<script>

let leftScore = 0;

let rightScore = 0;

let isLeftServing = true;


function update() {

  document.getElementById("leftScore").innerText = leftScore;

  document.getElementById("rightScore").innerText = rightScore;


  document.getElementById("left").classList.toggle("serving", isLeftServing);

  document.getElementById("right").classList.toggle("serving", !isLeftServing);


  document.getElementById("leftServe").innerText =

    isLeftServing ? serveText(leftScore) : "";


  document.getElementById("rightServe").innerText =

    !isLeftServing ? serveText(rightScore) : "";

}


function serveText(score) {

  return score % 2 === 0 ? "Serve: Right" : "Serve: Left";

}


document.getElementById("left").onclick = () => {

  leftScore++;

  isLeftServing = true;

  update();

};


document.getElementById("right").onclick = () => {

  rightScore++;

  isLeftServing = false;

  update();

};


function reset() {

  leftScore = 0;

  rightScore = 0;

  isLeftServing = true;

  update();

}


update();

</script>
